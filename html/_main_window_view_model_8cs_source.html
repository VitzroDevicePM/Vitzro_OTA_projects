34 public static void CopyDirectoryWithProgress(string source, string dest, IProgress<CopyProgress> progress, CancellationToken token)
35 {
36 // 전체 파일 목록 수집 (진행률 총량 계산)
37 var allFiles = System.IO.Directory.GetFiles(source, "*", SearchOption.AllDirectories);
38 var total = allFiles.Length;
39
40 // 대상 루트 생성
41 System.IO.Directory.CreateDirectory(dest);
42
43 // 먼저 모든 하위 폴더 생성
44 var allDirs = System.IO.Directory.GetDirectories(source, "*", SearchOption.AllDirectories);
45 foreach (var dir in allDirs)
46 {
47 token.ThrowIfCancellationRequested();
48 var rel = dir.Substring(source.Length).TrimStart(System.IO.Path.DirectorySeparatorChar);
49 var targetDir = System.IO.Path.Combine(dest, rel);
50 if (!System.IO.Directory.Exists(targetDir))
51 System.IO.Directory.CreateDirectory(targetDir);
52 }
53
54 // 파일 복사 + 진행률 보고
55 int done = 0;
56 foreach (var file in allFiles)
57 {
58 token.ThrowIfCancellationRequested();
59
60 var rel = file.Substring(source.Length).TrimStart(System.IO.Path.DirectorySeparatorChar);
61 var targetFile = System.IO.Path.Combine(dest, rel);
62 var targetDir = System.IO.Path.GetDirectoryName(targetFile);
63 if (!System.IO.Directory.Exists(targetDir))
64 System.IO.Directory.CreateDirectory(targetDir);
65
66 System.IO.File.Copy(file, targetFile, true); // overwrite
67 Logger.AppendLine(@"C:\Test", "파일 복사: " + Path.GetFileName(file));
68
69 done++;
70 if (progress != null)
71 {
72 int percent = (total == 0) ? 100 : (int)(done * 100.0 / total);
73 progress.Report(new CopyProgress
74 {
75 Done = done,
76 Total = total,
77 Percent = percent,
78 CurrentName = System.IO.Path.GetFileName(file)
79 });
80 }
81 }
82
83 // 빈 폴더만 있는 경우도 100% 보고
84 if (total == 0 && progress != null)
85 progress.Report(new CopyProgress { Done = 0, Total = 0, Percent = 100, CurrentName = "" });
86 }

156 private async Task UploadAsync()
157 {
158 _cts?.Dispose();
159 _cts = new CancellationTokenSource();
160
161 if (FilePath == "" || FilePath is null)
162 {
163 System.Windows.MessageBox.Show("경로를 선택 해주십시오", "Err");
164
165 return;
166 }
167
173 try
174 {
175 IsBusy = true;
176
177 IsIndeterminate = false;
178 BusyText = "복사 준비 중…";
179 ProgressValue = 0;
180
181 var source = FilePath;
182 var destRoot = @"C:\Test";
183 if (string.IsNullOrWhiteSpace(source) || !Directory.Exists(source))
184 throw new InvalidOperationException("유효한 폴더가 아닙니다.");
185
186 Directory.CreateDirectory(destRoot);
187
188 var dest = System.IO.Path.Combine(destRoot, new DirectoryInfo(source).Name);
189
190 var srcFull = System.IO.Path.GetFullPath(source).TrimEnd('\\');
191 var dstFull = System.IO.Path.GetFullPath(dest).TrimEnd('\\');
192 if (string.Equals(srcFull, dstFull, StringComparison.OrdinalIgnoreCase))
193 throw new InvalidOperationException("소스와 대상 경로가 같습니다.");
194
195 var progress = new Progress<CopyProgress>(p =>
196 {
197 ProgressValue = p.Percent;
198 BusyText = $"복사 중… {p.Done}/{p.Total} ({p.CurrentName})";
199 });
200
201 await Task.Run(() => CopyDirectoryWithProgress(source, dest, progress, _cts.Token), _cts.Token);
202 await Logger.AppendLineAsync(@"C:\Test", string.Format("복사 완료: '{0}' -> '{1}'", source, dest), _cts.Token);
203
204 BusyText = "복사 완료!";
205 }
206 catch (OperationCanceledException)
207 {
208 BusyText = "사용자 취소됨";
209 }
210 catch (Exception ex)
211 {
212 BusyText = $"오류: {ex.Message}";
213 }
214 finally
215 {
216 await Task.Delay(300);
217 IsBusy = false;
218 ProgressValue = 0;
219 IsIndeterminate = true;
220 }
221 }

269 private async Task ScanSubnetAsync()
270 {
271 _scanCts?.Dispose();
272 _scanCts = new CancellationTokenSource();
273
274 try
275 {
276 IsBusy = true;
277 IsIndeterminate = false;
278 ProgressValue = 0;

280
281 Hosts.Clear();
282
286 Progress<(int done, int total)> progress = new Progress<(int done, int total)>(p =>
287 {
288 ProgressValue = p.total == 0 ? 0 : (int)(p.done * 100.0 / p.total);
289 BusyText = $"스캔 중… {p.done}/{p.total}";
290 });
291
292 IList<HostRow> rows = await _scanner.ScanAsync(SubnetCidr, PingTimeoutMs, MaxConcurrency, progress, _scanCts.Token);
293 foreach (HostRow r in rows)
294 {
295 Hosts.Add(r);
296 }
297
298 BusyText = $"완료: {rows.Count}개 호스트";
299 }
300 catch (OperationCanceledException)
301 {
302 BusyText = "사용자 취소됨";
303 }
304 catch (Exception ex)
305 {
306 BusyText = $"오류: {ex.Message}";
307 }
308 finally
309 {
310 await Task.Delay(200);

312 IsIndeterminate = true;
