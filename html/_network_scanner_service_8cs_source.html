47 public string AutoDetectSubnetCidr()
48 {
49 // 기본 게이트웨이가 있는 Up NIC 우선 선택
50 foreach (var nic in NetworkInterface.GetAllNetworkInterfaces()
51 .Where(n => n.OperationalStatus == OperationalStatus.Up
52 && n.NetworkInterfaceType != NetworkInterfaceType.Loopback
53 && n.NetworkInterfaceType != NetworkInterfaceType.Tunnel))
54 {
55 var ipProps = nic.GetIPProperties();
56 if (ipProps.GatewayAddresses == null || ipProps.GatewayAddresses.Count == 0) continue;
57
58 var uni4 = ipProps.UnicastAddresses
59 .FirstOrDefault(u => u.Address.AddressFamily == AddressFamily.InterNetwork);
60 if (uni4 == null) continue;
61 if (uni4.IPv4Mask == null) continue;
62
63 var prefix = MaskToPrefixLength(uni4.IPv4Mask);
64 var network = GetNetworkAddress(uni4.Address, uni4.IPv4Mask);
65 return $"{network}/{prefix}";
66 }
67 // 실패 시 /24로 기본값
68 return "192.168.0.0/24";
69 }

71 public async Task<IList<HostRow>> ScanAsync(string cidr, int timeoutMs, int maxConcurrency, IProgress<(int done, int total)> progress, CancellationToken token)
72 {
73 var (network, prefix) = ParseCidr(cidr);
74 var hosts = EnumerateHosts(network, prefix).ToList();
75 int total = hosts.Count, done = 0;
76
77 var results = new List<HostRow>(total);
78 var sem = new SemaphoreSlim(maxConcurrency > 0 ? maxConcurrency : 128);
79
80 var tasks = hosts.Select(async ip =>
81 {
82 await sem.WaitAsync(token);
83 try
84 {
85 var row = await ProbeAsync(ip, timeoutMs, token);
86 lock (results) results.Add(row);
87 }
88 finally
89 {
90 var curr = Interlocked.Increment(ref done);
91 progress?.Report((curr, total));
92 sem.Release();
93 }
94 }).ToList();
95
96 await Task.WhenAll(tasks);
97 // IP 정렬
98 return results.OrderBy(r => IPToUInt(IPAddress.Parse(r.IP))).ToList();
99 }

108 private static (IPAddress network, int prefix) ParseCidr(string cidr)
109 {
110 // 허용: "192.168.0.0/24" 또는 "192.168.0.15/24"
111 var parts = cidr.Split('/');
112 if (parts.Length != 2) throw new ArgumentException("CIDR 형식이 올바르지 않습니다. 예: 192.168.0.0/24");
113 var ip = IPAddress.Parse(parts[0]);
114 int prefix = int.Parse(parts[1]);
115 var mask = PrefixToMask(prefix);
116 var net = GetNetworkAddress(ip, mask);
117 return (net, prefix);
118 }

127 private static IEnumerable<IPAddress> EnumerateHosts(IPAddress network, int prefix)
128 {
129 if (prefix < 0 || prefix > 32) throw new ArgumentOutOfRangeException(nameof(prefix));
130 var mask = PrefixToMask(prefix);
131
132 uint net = IPToUInt(GetNetworkAddress(network, mask));
133 uint bcast = net | ~IPToUInt(mask);
134
135 // /31,/32 처리는 호스트 없음 처리
136 if (prefix >= 31) yield break;
137
138 for (uint u = net + 1; u < bcast; u++)
139 yield return UIntToIP(u);
140 }

150 private static async Task<HostRow> ProbeAsync(IPAddress ip, int timeoutMs, CancellationToken token)
151 {
152 var ping = new Ping();
153 try
154 {
155 var sw = System.Diagnostics.Stopwatch.StartNew();
156 var reply = await ping.SendPingAsync(ip, timeoutMs);
157 sw.Stop();
158
159 if (reply.Status == IPStatus.Success)
160 {
161 string hostname = null;
162 try
163 {
164 // 역방향 DNS (느릴 수 있으니 필요 시 주석처리)
165 var entry = await Dns.GetHostEntryAsync(ip);
166 hostname = entry?.HostName;
167 }
168 catch { /* DNS 실패 무시 */ }
169
170 return new HostRow
171 {
172 IP = ip.ToString(),
173 Hostname = hostname,
174 LatencyMs = sw.ElapsedMilliseconds,
175 IsAlive = true
176 };
177 }
178 }
179 catch { /* ping 예외 무시 */ }
180
181 return new HostRow
182 {
183 IP = ip.ToString(),
184 Hostname = null,
185 LatencyMs = null,
186 IsAlive = false
187 };
188 }

211 private static IPAddress PrefixToMask(int prefix)
212 {
213 uint mask = prefix == 0 ? 0u : 0xffffffffu << (32 - prefix);
214 return UIntToIP(mask);
215 }

224 private static int MaskToPrefixLength(IPAddress mask)
225 {
226 uint m = IPToUInt(mask);
227 int count = 0;
228 for (int i = 31; i >= 0; i--)
229 if (((m >> i) & 1) == 1) count++;
230 else break;
231 return count;
232 }

241 private static uint IPToUInt(IPAddress ip)
242 {
243 var bytes = ip.GetAddressBytes(); // big-endian
244 if (BitConverter.IsLittleEndian) Array.Reverse(bytes);
245 return BitConverter.ToUInt32(bytes, 0);
246 }

255 private static IPAddress UIntToIP(uint value)
256 {
257 var bytes = BitConverter.GetBytes(value);
258 if (BitConverter.IsLittleEndian) Array.Reverse(bytes);
259 return new IPAddress(bytes);
